# ROLLBACK COMMANDS - Use if you lose access
# Copy these commands to RouterOS Terminal if admin access is lost

# ============================================================================
# EMERGENCY: Restore Admin Network Access
# ============================================================================

# If you lost access, connect via:
# - Winbox (port 8291)
# - WebFig (http://10.1.1.1)
# - SSH (port 22)

# Then run these commands:

# 1. Re-enable the general LAN->WAN rule (temporary)
/ip firewall filter set [find chain=forward in-interface-list=LAN out-interface-list=WAN] disabled=no

# 2. Ensure admin network rule exists
/ip firewall filter add chain=forward action=accept src-address=10.1.1.0/24 out-interface-list=WAN comment="allow-admin-network-full-access" place-before=0

# 3. Verify admin network rule is active
/ip firewall filter print where src-address=10.1.1.0/24

# ============================================================================
# Alternative: Disable hotspot blocking rule
# ============================================================================

# If admin access is blocked, disable the hotspot blocking rule:
/ip firewall filter set [find comment="block-unauthenticated-hotspot"] disabled=yes

# ============================================================================
# Verify Current Rules
# ============================================================================

# Check all forward rules
/ip firewall filter print where chain=forward

# Check admin network rules
/ip firewall filter print where src-address=10.1.1.0/24

# Check hotspot rules
/ip firewall filter print where in-interface=bridge-hotspot
